create_flag_panel <- function(flag_name, panel_label) {
  d <- anomaly_plot_data %>% filter(flag_type == flag_name)
  ggplot(d, aes(x = jurisdiction, y = count, fill = metric)) +
    geom_bar(stat = "identity", position = "stack", color = "black",
             linewidth = 0.2, width = 0.8) +
    shared_fill +
    scale_x_discrete(limits = all_jurisdictions, drop = FALSE) +
    scale_y_continuous(expand = expansion(mult = c(0, 0.1))) +
    labs(title = paste0(panel_label, " ", flag_panel_labels[flag_name]), y = "Count") +
    theme_panel
}

growth_data_yearly <- ndis_clean_wide %>%
  mutate(year = year(capture_datetime)) %>%
  group_by(jurisdiction, year) %>%
  summarise(
    offender_profiles = max(offender_profiles, na.rm = TRUE),
    arrestee = max(arrestee, na.rm = TRUE),
    forensic_profiles = max(forensic_profiles, na.rm = TRUE),
    investigations_aided = max(investigations_aided, na.rm = TRUE),
    .groups = "drop"
  ) %>%
  # Handle Inf values from max on empty groups
  mutate(across(c(offender_profiles, arrestee, forensic_profiles, investigations_aided), 
                ~ifelse(is.infinite(.), NA, .))) %>%
  group_by(year) %>%
  summarise(

    arrestee_total = sum(arrestee, na.rm = TRUE),
    forensic_total = sum(forensic_profiles, na.rm = TRUE),
    investigations_total = sum(investigations_aided, na.rm = TRUE),
    n_jurisdictions = n(),
    .groups = 'drop'
  ) %>%
  mutate(
    total_profiles = offender_total + arrestee_total + forensic_total,
    date = as.Date(paste0(year, "-06-01"))
  )
